SELECT
  FIRST_NAME,
  LAST_NAME,
  PHONE
FROM STUDENT
WHERE REGISTRATION_DATE = TO_DATE('2/23/2007', 'MM-DD-YYYY')
ORDER BY LAST_NAME, FIRST_NAME;

--2
SELECT
  COURSE_NO,
  SECTION_ID,
  TO_CHAR(START_DATE_TIME, 'DD-MON-RR') AS START_DATE_TIME
FROM SECTION
WHERE LOCATION = 'L211'
ORDER BY START_DATE_TIME;

--3
SELECT
  COURSE_NO,
  SECTION_ID,
  TO_CHAR(START_DATE_TIME, 'DD-MON-RR') AS START_DATE_TIME,
  INSTRUCTOR_ID,
  CAPACITY
FROM SECTION
-- WHERE TO_CHAR(START_DATE_TIME, 'fmMonth YYYY') = 'April 2007'
WHERE 0 < INSTR(START_DATE_TIME, 'APR')
ORDER BY START_DATE_TIME ASC, COURSE_NO ASC;

--4
SELECT
  STUDENT_ID,
  SECTION_ID,
  FINAL_GRADE
FROM ENROLLMENT
WHERE FINAL_GRADE IS NOT NULL AND TO_CHAR(ENROLL_DATE, 'Mon-YYYY') = 'Jan-2007';

--5

SELECT TO_CHAR(TO_DATE('07-JAN-13', 'DD-Mon-YY') + INTERVAL '98' DAY,'DD-MON-YY') SEMESTER_END_DATE
FROM DUAL;

--6
SELECT Course_No
FROM Section
WHERE SUBSTR(LOCATION, 1, 1) = 'M' AND MOD(SUBSTR(LOCATION, 4, 1), 2) = 1
ORDER BY Course_No;

--7
SELECT
  LAST_NAME,
  FIRST_NAME
FROM STUDENT
WHERE ZIP = '11214' AND REGISTRATION_DATE - STUDENT.CREATED_DATE > 2
ORDER BY LAST_NAME;

--8
SELECT
  FIRST_NAME,
  LAST_NAME,
  Round((sysdate - REGISTRATION_DATE) / 365.25, 2) AS YEARS
FROM STUDENT
WHERE SUBSTR(PHONE, 1, 3) = '203'
ORDER BY YEARS DESC, FIRST_NAME;

--9
SELECT DISTINCT to_char(START_DATE_TIME, 'HH:MI') AS STARTTIME
FROM SECTION
ORDER BY STARTTIME;

--10
SELECT
  STUDENT_ID,
  SECTION_ID
FROM ENROLLMENT
WHERE To_char(ENROLL_DATE, 'HH:MI') = '10:20';
